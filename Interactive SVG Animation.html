// SVG Animation with JavaScript
const svg = document.querySelector('.animated-svg');

// Circle animation
function animateCircle() {
    const circle = svg.querySelector('.animated-circle');
    circle.style.transition = 'stroke-dashoffset 2s ease-in-out';
    circle.style.strokeDashoffset = '0';
}

// Path animation
function animatePath() {
    const path = svg.querySelector('.animated-path');
    path.style.transition = 'stroke-dashoffset 2s ease-in-out';
    path.style.strokeDashoffset = '0';
}

// Interactive elements
document.querySelectorAll('.interactive-rect').forEach(rect => {
    rect.addEventListener('mouseenter', () => {
        rect.style.transform = 'scale(1.2)';
        rect.style.transition = 'transform 0.3s ease';
    });
    
    rect.addEventListener('mouseleave', () => {
        rect.style.transform = 'scale(1)';
    });
});

// Reset animations
function resetAnimations() {
    const circle = svg.querySelector('.animated-circle');
    const path = svg.querySelector('.animated-path');
    
    circle.style.transition = 'none';
    path.style.transition = 'none';
    
    circle.style.strokeDashoffset = '377';
    path.style.strokeDashoffset = '300';
}
