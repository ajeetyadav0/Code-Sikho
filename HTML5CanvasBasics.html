HTML5 Canvas Basics

Drawing Shapes

var canvas = document.getElementById("myCanvas"); var ctx = canvas.getContext("2d"); // Rectangle ctx.fillStyle = "#FF0000"; ctx.fillRect(10, 10, 50, 50); // Circle ctx.beginPath(); ctx.arc(100, 60, 20, 0, 2 * Math.PI); ctx.fillStyle = "#00FF00"; ctx.fill();

Drawing Text

ctx.font = "30px Arial"; ctx.fillStyle = "blue"; ctx.fillText("Hello World", 10, 50);

Drawing Lines

ctx.beginPath(); ctx.moveTo(20, 20); ctx.lineTo(200, 100); ctx.strokeStyle = "red"; ctx.stroke();
